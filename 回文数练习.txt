package com.lanqiao.one;

public class 回文数练习 {

	public static void main(String[] args) {
/*	问题描述
		　　1221是一个非常特殊的数，它从左边读和从右边读是一样的，编程求所有这样的四位十进制数。
		  输出格式
		  　　按从小到大的顺序输出满足条件的四位十进制数。*/
		int sum=0;
		
	for(int i=1000;i<10000;i++){
		int a=i/1000;
		int b=i%1000/100;
		int c=i%100/10;
		int d=i%10;
		sum=a+b*10+c*100+d*1000;
		if(sum==i){
			System.out.println(i);
		}
		
	}
	}

}
